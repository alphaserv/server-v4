map

function map(t,
f)

Remaps a table (array).  Usage -

local a = { 1, 2, 3, 4, 5 }
local b = table.map(a, function(i) return tostring(i) end)
-- b is now a table of strings

Parameters

tThe table to remap.
fA function taking original element and returning remapped.

Returns

A remapped table.

function copy(t)

Copies a table.  Remember, it does not take care of copying in case member is a table.  Usage -

local a = { 5, 10, 15 }
local b = table.copy(a)
-- b is now a copy of a

Parameters

tThe table to copy.

Returns

Copied table.

filter_dict

function filter_dict(t,
f)

Filters a table (dictionary).  Usage -

local a = { a = 5, b = 10 }
local b = table.filter_dict(a, function (k, v)
    return ((v <= 5) and true or false)
end)
-- b now contains just "a"

Parameters

tThe table to filter.
fA function taking key and value and returning true if current element should be included in new table and false otherwise.

Returns

A filtered table.

function filter_array(t,
f)

Filters a table (array).  Usage -

local a = { 5, 10, 15 }
local b = table.filter_array(a, function (i, v)
    return ((i <= 2) and true or false)
end)
-- b is empty

Parameters

tThe table to filter.
fA function taking index and value and returning true if current element should be included in new table and false otherwise.

Returns

A filtered table.

pop

function pop(t,
p)

Pops item from end of the table or from specific position and returns it.  Usage -

local a = { 5, 10, 15, 20, 25, 30, 35, 40 }
local b = table.pop(a)
-- b is now 40, a is { 5, 10, 15, 20, 25, 30, 35 }
b = table.pop(a, 3)
-- b is now 15, a is { 5, 10, 20, 25, 30, 35 }

Parameters

tThe table to pop item from.
pOptional argument specifying index, it’s the end of the table if not specified.

Returns

The popped item.

function sum(t)

Sums a table of numbers.  Usage -

local a = { 5, 10, 15 }
local b = table.sum(a)
-- b is now 5 + 10 + 15, that is 30

Parameters

tThe table of numbers to sum.

Returns

The sum.
